/********************************************************************** * Homework II * * * * Prompt: Given a set S, return the power set P(S), which is * * a set of all subsets of S. * * * * Input: A String * * Output: An Array with the power set of the input string * * * * Example: S = "abc", P(S) = ['', 'a', 'b','c','ab','ac','bc','abc']* * * * * Note: There should not be any duplicates in the power set * * ('ab' and 'ba' are considered the same), and it will always * * begin with an empty string ('') * * * **********************************************************************/ var powerSet = function(str){ } 

function powerSet(input){
    var results = [];
        function traverse(build, depth)
            {
                if(depth == input.length)
                {

                    results.push(build)
                    return
                }

                traverse(build, depth+1);
                traverse(build+input[depth], depth+1)
            }
    traverse("", 0)
    return results;
}

str = "abc";
powerSet(str);
